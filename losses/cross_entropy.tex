\section{Entropy}

$$
E(t) = \sum_{i=1}^{M}t_i\log(\frac{1}{t_i})
$$


\section{Cross Entropy}

$$
    L(p, t) = \sum_{i=1}^{M} t_i \log(\frac{1}{p_i})
$$

\section{Binary Cross Entropy}

$$
    L(p, t) = \sum_{i=0}^{1} t_i \log(\frac{1}{p_i}) = t_0\log(\frac{1}{p_0}) + t_1\log(\frac{1}{p_1})
$$
where
$$
    t_1 = 1 - t_0
$$
